WhatsApp Bot — Baileys (Node.js)

Upload a file directly from your bot and send the URL or the media in a message. Works with SHINIGAMI-MD, INCONNU XD, and any Baileys-based bot.

UPLOAD.JS const FormData = require('form-data'); const axios = require('axios'); const fs = require('fs'); async function uploadToIncoHost(filePath) { const form = new FormData(); form.append('file', fs.createReadStream(filePath)); const { data } = await axios.post( 'https://your-app.onrender.com/api/upload', form, { headers: form.getHeaders() } ); return data.url; }

Discord Bot — discord.js v14

Upload attachments from Discord commands and store them permanently. Send back the URL or embed the image directly in a reply.

SLASH COMMAND const { SlashCommandBuilder } = require('discord.js'); const FormData = require('form-data'); const axios = require('axios'); module.exports = { data: new SlashCommandBuilder() .setName('host') .setDescription('Upload a file to INCO.HOST') .addAttachmentOption(opt => opt.setName('file').setDescription('File to upload').setRequired(true) ), async execute(interaction) { const att = interaction.options.getAttachment('file'); const resp = await axios.get(att.url, { responseType: 'arraybuffer' }); const form = new FormData(); form.append('file', Buffer.from(resp.data), att.name); const { data } = await axios.post( 'https://your-app.onrender.com/api/upload', form, { headers: form.getHeaders() } ); await interaction.reply({ content: `✅ \`${data.url}\`` }); } };

Telegram Bot — node-telegram-bot-api

Upload photos or videos sent to your Telegram bot and store them permanently with a shareable link.

TELEGRAM BOT const TelegramBot = require('node-telegram-bot-api'); const FormData = require('form-data'); const axios = require('axios'); const bot = new TelegramBot('YOUR_TOKEN', { polling: true }); bot.on('photo', async (msg) => { const fileId = msg.photo.at(-1).file_id; const fileUrl = await bot.getFileLink(fileId); const buf = (await axios.get(fileUrl, { responseType: 'arraybuffer' })).data; const form = new FormData(); form.append('file', Buffer.from(buf), 'photo.jpg'); const { data } = await axios.post( 'https://your-app.onrender.com/api/upload', form, { headers: form.getHeaders() } ); bot.sendMessage(msg.chat.id, `🔗 ${data.url}`); });

API REFERENCE

Complete REST API — integrate from any language or platform

POST /api/upload
Upload a single file. Field name: file. Returns direct URL + delete key.
curl -X POST https://your-app.com/api/upload -F "file=@image.jpg"
POST /api/upload/multi
Upload up to 10 files at once. Field name: files (repeated). Returns array of URLs.
curl -X POST https://your-app.com/api/upload/multi -F "files=@a.jpg" -F "files=@b.mp4"
GET /api/file/:id
Get metadata for a file by its ID (no delete key exposed).
GET /api/recent
Returns last 20 uploaded files with URLs and metadata.
GET /api/stats
Returns total file count, size breakdown by type, and platform info.
DELETE /api/delete/:id?key=deleteKey
Permanently delete a file. Requires the deleteKey returned at upload time.
curl -X DELETE https://your-app.com/api/delete/uuid?key=deleteKey
GET /api/info
Returns detected platform, base URL, and file size limits.
